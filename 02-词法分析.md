```cpp
static char* readFile(const char* path) {
  FILE* file = fopen(path, "rb");

  fseek(file, 0L, SEEK_END);
  size_t fileSize = ftell(file);
  rewind(file);

  char* buffer = (char*)malloc(fileSize + 1);
  size_t bytesRead = fread(buffer, sizeof(char), fileSize, file);
  buffer[bytesRead] = '\0';

  fclose(file);
  return buffer;
}
```

```cpp
static Token makeToken(TokenType type) {
  Token token;
  token.type = type;
  token.start = scanner.start;
  token.length = (int)(scanner.current - scanner.start);
  token.line = scanner.line;

  return token;
}

static Token errorToken(const char* message) {
  Token token;
  token.type = ILLEGAL;
  token.start = message;
  token.length = (int)strlen(message);
  token.line = scanner.line;

  return token;
}
```

```cpp
static char advance() {
  scanner.current++;
```

`match`的代码

```cpp
static bool match(char expected) {
  if (isAtEnd()) return false;
  if (*scanner.current != expected) return false;

  scanner.current++;
  return true;
}
```

跳过空白字符

```cpp
static void skipWhitespace() {
  for (;;) {
    char c = peek();
    switch (c) {
      case ' ':
      case '\r':
      case '\t':
        advance();
        break;

      case '\n':
        scanner.line++;
        advance();
        break;

      default:
        return;
    }
  }
}
```

```cpp
static Token string() {
  while (peek() != '"' && !isAtEnd()) {
    if (peek() == '\n') scanner.line++;
    advance();
  }

  if (isAtEnd()) return errorToken("Unterminated string.");

  // The closing quote.
  advance();
  return makeToken(STRING);
}
```

```cpp
static TokenType checkKeyword(int start, int length,
    const char* rest, TokenType type) {
  if (scanner.current - scanner.start == start + length &&
      memcmp(scanner.start + start, rest, length) == 0) {
    return type;
  }

  return IDENT;
}

static TokenType identifierType() {
  switch (scanner.start[0]) {
    case 'e': return checkKeyword(1, 3, "lse", ELSE);
    case 'i': return checkKeyword(1, 1, "f", IF);
    case 'r': return checkKeyword(1, 5, "eturn", RETURN);
    case 'l': return checkKeyword(1, 2, "et", LET);
    case 't': return checkKeyword(1, 3, "ure", TRUE);
    case 'f':
      if (scanner.current - scanner.start > 1) {
        switch (scanner.start[1]) {
          case 'a': return checkKeyword(2, 3, "lse", FALSE);
          case 'n': return FUNCTION;
        }
      }
      break;
  }
  return IDENT;
}
```
